### Feature 1 — Wire Image chip to backend OCR `(1.5 hrs)`

The backend `/api/parse-image` endpoint will be built by Member 2. But you can build the frontend independently using a mock response.

**`frontend/src/routes/chat.tsx`** — wire the Image chip button:

```tsx
// Add a hidden file input ref
const imageInputRef = useRef<HTMLInputElement>(null)

// Wire the Image chip button
<button onClick={() => imageInputRef.current?.click()}>
  <ImageIcon className="h-3.5 w-3.5" /> Image
</button>

<input
  ref={imageInputRef}
  type="file"
  accept="image/*"
  className="hidden"
  onChange={async (e) => {
    const file = e.target.files?.[0]
    if (!file) return
    setPhase("thinking")
    setMessages(m => [...m, { role: "user", text: `📷 Uploaded: ${file.name}` }])

    const formData = new FormData()
    formData.append("image", file)
    if (budgetInput) formData.append("budget_inr", budgetInput)

    try {
      const res = await fetch("/api/parse-image", { method: "POST", body: formData })
      if (!res.ok) throw new Error("Image parsing failed")
      const data = await res.json()
      // If backend returns extracted_text, submit it to /api/parse
      if (data.extracted_text) {
        setText(data.extracted_text)
        // Auto-submit
        onSubmit()
      }
    } catch (err: any) {
      setErrorMsg(err.message)
      setPhase("idle")
    }
  }}
/>
```

No backend dependency — works with mock mode. If `/api/parse-image` isn't ready, it'll show an error gracefully.

### Feature 3 — Preferences persistence in localStorage `(1 hr)`

Wire the existing Preferences page to save/load from localStorage.

**`frontend/src/lib/preferences.ts`** (new file):
```ts
export interface UserPreferences {
  dietary: "any" | "veg" | "vegan" | "jain"
  preferredBrands: string[]
  budgetStyle: "value" | "balanced" | "premium"
}

const STORAGE_KEY = "needspeak-preferences"

const DEFAULTS: UserPreferences = {
  dietary: "any",
  preferredBrands: [],
  budgetStyle: "balanced",
}

export function loadPreferences(): UserPreferences {
  try {
    const raw = localStorage.getItem(STORAGE_KEY)
    if (!raw) return DEFAULTS
    return { ...DEFAULTS, ...JSON.parse(raw) }
  } catch {
    return DEFAULTS
  }
}

export function savePreferences(prefs: UserPreferences): void {
  localStorage.setItem(STORAGE_KEY, JSON.stringify(prefs))
}
```

Then in `/preferences` route, wire the form to `loadPreferences()` on mount and `savePreferences()` on save.

In `chat.tsx` `onSubmit`, include preferences in the request body:

if (prefs.preferredBrands.length) body.preferred_brands = prefs.preferredBrands
if (prefs.budgetStyle !== "balanced") body.budget_style = prefs.budgetStyle

### Feature 1 — Gemini Vision image OCR endpoint `(2 hrs)`

Create the endpoint that Member 1's Image chip will call.

**`backend/app/ingestion/image_input.py`** (new file):
```python
import base64
from app.pipeline.gemini_client import get_gemini_client
from app import config

def extract_text_from_image(image_bytes: bytes, mime_type: str) -> str:
    """Use Gemini Vision to extract shopping items from an image."""
    client = get_gemini_client()
    encoded = base64.b64encode(image_bytes).decode("utf-8")

    response = client.models.generate_content(
        model=config.GEMINI_MODEL_ID,
        contents=[{
            "role": "user",
            "parts": [
                {"inline_data": {"mime_type": mime_type, "data": encoded}},
                {"text": (
                    "Look at this image carefully. Extract ALL food items, ingredients, "
                    "grocery items, or shopping items visible. Include quantities if readable. "
                    "Return as a plain comma-separated list. "
                    "Example: milk 2L, basmati rice 1kg, onions 500g, Maggi noodles 4 packs"
                )}
            ]
        }]
    )
    return response.text.strip() if response.text else ""
```

**In `backend/app/main.py`** — add at the bottom before health check:
```python
@app.post("/api/parse-image")
async def parse_image(request: Request, image: UploadFile = File(...)):
    """Accept an image, extract items via Gemini Vision, return extracted text."""
    mock_mode = getattr(request.state, "mock_mode", False) or config.MOCK_MODE
    
    image_bytes = await image.read()
    mime_type = image.content_type or "image/jpeg"
    
    if len(image_bytes) < 1000:
        raise HTTPException(status_code=400, detail={"message": "Image too small."})
    
    logger.info(f"[parse-image] Received {len(image_bytes)} bytes, mime={mime_type}")
    
    if mock_mode:
        return {"extracted_text": "milk 1L, bread 1 pack, eggs 12, onions 1kg, tomatoes 500g"}
    
    try:
        from app.ingestion.image_input import extract_text_from_image
        extracted = extract_text_from_image(image_bytes, mime_type)
        
        if not extracted:
            raise HTTPException(status_code=400, detail={"message": "No items found in image."})
        
        logger.info(f"[parse-image] Extracted: {extracted[:100]}")
        return {"extracted_text": extracted}
    
    except HTTPException:
        raise
    except Exception as e:
        error_str = str(e)
        if "429" in error_str or "RESOURCE_EXHAUSTED" in error_str:
            raise HTTPException(status_code=429, detail={"message": "Rate limit. Try again in a minute."})
        raise HTTPException(status_code=500, detail={"message": f"Image parsing failed: {error_str}"})
```

No frontend dependency. Self-contained. Testable via curl:
```bash
curl -X POST http://localhost:8000/api/parse-image -F "image=@photo.jpg"
```

### Feature 1 — Real CompareCart re-run `(2.5 hrs)`

When user moves the budget slider on the ReviewCart page, re-call `/api/parse` with the new budget and show a diff.

**`frontend/src/lib/cart-diff.ts`** (new file):
```ts
export interface CartDiff {
  added: any[]      // items in new but not old
  removed: any[]    // items in old but not new
  swapped: { old: any; new: any }[]  // same item name, different sku/price
  unchanged: any[]
}

export function diffCarts(oldCart: any[], newCart: any[]): CartDiff {
  const oldMap = new Map(oldCart.map(i => [i.name.toLowerCase(), i]))
  const newMap = new Map(newCart.map(i => [i.name.toLowerCase(), i]))

  const added: any[] = []
  const removed: any[] = []
  const swapped: { old: any; new: any }[] = []
  const unchanged: any[] = []

  for (const [name, item] of newMap) {
    const oldItem = oldMap.get(name)
    if (!oldItem) {
      added.push(item)
    } else if (oldItem.sku !== item.sku || oldItem.price_per_unit_inr !== item.price_per_unit_inr) {
      swapped.push({ old: oldItem, new: item })
    } else {
      unchanged.push(item)
    }
  }

  for (const [name, item] of oldMap) {
    if (!newMap.has(name)) removed.push(item)
  }

  return { added, removed, swapped, unchanged }
}
```

**In `frontend/src/routes/cart.$id.tsx`** — on budget slider change:
```tsx
const [diffResult, setDiffResult] = useState<CartDiff | null>(null)
const [comparing, setComparing] = useState(false)

const runCompare = async (newBudget: number) => {
  setComparing(true)
  try {
    const res = await fetch("/api/parse", {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({
        content: session.originalInput || session.context_summary,
        input_type: "text",
        budget_inr: newBudget,
      }),
    })
    const data = await res.json()
    const newCart = (data.intents ?? []).flatMap((g: any) => g.cart ?? [])
    setDiffResult(diffCarts(session.cart, newCart))
  } catch { /* ignore */ }
  setComparing(false)
}

// Show diff below slider:
{diffResult && (
  <div className="mt-4 space-y-2 text-xs">
    {diffResult.added.map(i => (
      <div key={i.sku} className="text-success">+ {i.name} (₹{i.price_per_unit_inr})</div>
    ))}
    {diffResult.removed.map(i => (
      <div key={i.sku} className="text-destructive">− {i.name}</div>
    ))}
    {diffResult.swapped.map(s => (
      <div key={s.new.sku} className="text-brand">↔ {s.old.name} → {s.new.name} (save ₹{(s.old.price_per_unit_inr - s.new.price_per_unit_inr).toFixed(0)})</div>
    ))}
  </div>
)}
```

No backend change. Uses existing `/api/parse` endpoint.

const BADGE_RULES: Record<string, { label: string; color: string }> = {
  "VEG": { label: "Seasonal", color: "text-amber-600 bg-amber-50" },
  "CHKN": { label: "Low Stock", color: "text-red-600 bg-red-50" },
  "ICE": { label: "Frozen", color: "text-blue-600 bg-blue-50" },
}

export function getItemBadge(sku: string): { label: string; color: string } | null {
  for (const [prefix, badge] of Object.entries(BADGE_RULES)) {
    if (sku?.startsWith(prefix)) return badge
  }
  return null
}
